250 typedef struct {
251  uint32_t Timestamp; //set by parse/send network routines
252  //VER is implicit = 1
253  //TKL (Token Length) is calculated dynamically
254  CoAP_MessageType_t Type; // [1] T
255  CoAP_MessageCode_t Code; // [1] Code
256  uint16_t MessageID; // [2] Message ID (maps ACK msg to coresponding CON msg)
257  uint16_t PayloadLength; // [2]
258  uint16_t PayloadBufSize; // [2] size of allocated msg payload buffer
259  CoAP_Token_t Token; // [9] Token (1 byte Length + up to 8 Byte for the token content)
260  CoAP_option_t *pOptionsList; // [4] linked list of Options
261  uint8_t *Payload; // [4] MUST be last in struct! Because of mem allocation scheme which tries to allocate message mem and payload mem in ONE big data chunk
262 
263  struct CoAP_Res *pResource; // Pointer the the resource this message is intended for.
264 } CoAP_Message_t; //total of 25 Bytes
